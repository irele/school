#include "Header.h"

double y(double x, int n)
{
	double rez = 1;
	if (x > 0)
		for (int i = 1; i <= n; i++)
			for (int j = 0; j < n; j++)
				rez *= x - i - j;
	else
	{
		rez = 0;
		for (int i = 0; i <= n; i++)
			rez += (x - i)*(x - i);
	}
	return rez;
}

int count(funcT f, double a, double b, double h, int n)
{
	int N = 0;
	while (a <= b)
	{
		if (f(a, n) == 0)
			N++;
		a += h;
	}
	return N;
}
